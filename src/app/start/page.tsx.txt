import { redirect } from "next/navigation";
import { getSupabaseServer } from "@/lib/supabase/server";
import { PetitionForm } from "@/components/petition/petition-form";

export const dynamic = "force-dynamic";

export default async function StartPetitionPage() {
  const supabase = getSupabaseServer();

  // Check if user is authenticated
  const {
    data: { user },
  } = await supabase.auth.getUser();

  if (!user) {
    redirect("/auth?redirect=/start");
  }

  return (
    <div className="container mx-auto px-4 py-16 max-w-3xl">
      <h1 className="text-4xl font-bold mb-8 text-gray-900">
        Start a Petition
      </h1>
      <p className="text-gray-600 mb-8">
        Create a petition on an issue you care about. Be clear, specific, and
        passionate about your cause.
      </p>
      <PetitionForm />
    </div>
  );
}
