import { Resend } from "resend";

const resendApiKey = process.env.RESEND_API_KEY;
const fromEmail = process.env.EMAIL_FROM || "voice@myindia.ai";
const siteUrl = process.env.NEXT_PUBLIC_SITE_URL || "https://myindia-staging.vercel.app";

if (!resendApiKey) {
  console.warn("⚠️  RESEND_API_KEY not configured. Emails will fail.");
}

const resend = resendApiKey ? new Resend(resendApiKey) : null;

export async function sendSignatureConfirmationEmail(
  to: string,
  petitionTitle: string,
  confirmToken: string
): Promise<void> {
  if (!resend) {
    console.error("Cannot send email: Resend not configured");
    throw new Error("Email service not configured");
  }

  const confirmUrl = `${siteUrl}/api/signatures/confirm?token=${confirmToken}`;

  try {
    await resend.emails.send({
      from: fromEmail,
      to,
      subject: `Confirm your signature: ${petitionTitle}`,
      html: `
        <!DOCTYPE html>
        <html>
          <head>
            <meta charset="utf-8">
            <style>
              body { font-family: Arial, sans-serif; line-height: 1.6; color: #333; }
              .container { max-width: 600px; margin: 0 auto; padding: 20px; }
              .button { 
                display: inline-block; 
                padding: 12px 24px; 
                background-color: #000; 
                color: #fff; 
                text-decoration: none; 
                border-radius: 4px; 
                margin: 20px 0;
              }
              .footer { margin-top: 30px; font-size: 12px; color: #666; }
            </style>
          </head>
          <body>
            <div class="container">
              <h2>Confirm Your Signature</h2>
              <p>Thank you for supporting: <strong>${petitionTitle}</strong></p>
              <p>Please confirm your signature by clicking the button below:</p>
              <a href="${confirmUrl}" class="button">Confirm Signature</a>
              <p>Or copy and paste this link into your browser:</p>
              <p style="word-break: break-all; color: #666;">${confirmUrl}</p>
              <div class="footer">
                <p>If you did not sign this petition, you can safely ignore this email.</p>
                <p>This link will expire in 24 hours.</p>
              </div>
            </div>
          </body>
        </html>
      `,
    });
  } catch (error) {
    console.error("Failed to send confirmation email:", error);
    throw new Error("Failed to send confirmation email");
  }
}
