"use client";

import { useState } from "react";

interface SignatureFormProps {
  petitionId: string;
  petitionTitle: string;
}

export function SignatureForm({ petitionId, petitionTitle }: SignatureFormProps) {
  const [loading, setLoading] = useState(false);
  const [success, setSuccess] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const [formData, setFormData] = useState({
    name: "",
    email: "",
    comment: "",
  });

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    setError(null);
    setSuccess(false);

    try {
      const response = await fetch("/api/signatures", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          petitionId,
          name: formData.name,
          email: formData.email,
          comment: formData.comment || undefined,
        }),
      });

      const data = await response.json();

      if (!response.ok) {
        throw new Error(data.error || "Failed to submit signature");
      }

      setSuccess(true);
      setFormData({ name: "", email: "", comment: "" });
    } catch (err: any) {
      setError(err.message);
    } finally {
      setLoading(false);
    }
  };

  if (success) {
    return (
      <div className="p-6 bg-green-50 border border-green-200 rounded-lg">
        <h3 className="text-lg font-semibold text-green-900 mb-2">
          Check Your Email!
        </h3>
        <p className="text-green-800">
          We've sent a confirmation link to your email address. Please click the link
          to confirm your signature.
        </p>
      </div>
    );
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4 bg-gray-50 p-6 rounded-lg">
      <h3 className="text-xl font-semibold mb-4">Sign This Petition</h3>

      <div>
        <label htmlFor="name" className="block text-sm font-medium mb-1">
          Your Name *
        </label>
        <input
          id="name"
          type="text"
          required
          className="w-full px-4 py-2 border rounded-md focus:ring-2 focus:ring-black focus:outline-none"
          value={formData.name}
          onChange={(e) => setFormData({ ...formData, name: e.target.value })}
          placeholder="Full Name"
        />
      </div>

      <div>
        <label htmlFor="email" className="block text-sm font-medium mb-1">
          Your Email *
        </label>
        <input
          id="email"
          type="email"
          required
          className="w-full px-4 py-2 border rounded-md focus:ring-2 focus:ring-black focus:outline-none"
          value={formData.email}
          onChange={(e) => setFormData({ ...formData, email: e.target.value })}
          placeholder="you@example.com"
        />
        <p className="text-xs text-gray-600 mt-1">
          We'll send you a confirmation email
        </p>
      </div>

      <div>
        <label htmlFor="comment" className="block text-sm font-medium mb-1">
          Comment (Optional)
        </label>
        <textarea
          id="comment"
          rows={3}
          className="w-full px-4 py-2 border rounded-md focus:ring-2 focus:ring-black focus:outline-none"
          value={formData.comment}
          onChange={(e) => setFormData({ ...formData, comment: e.target.value })}
          placeholder="Why are you signing this petition?"
        />
      </div>

      {error && (
        <div className="p-3 bg-red-50 border border-red-200 rounded-md text-red-700 text-sm">
          {error}
        </div>
      )}

      <button
        type="submit"
        disabled={loading}
        className="w-full bg-black text-white py-3 rounded-md hover:bg-gray-800 disabled:opacity-50 disabled:cursor-not-allowed font-medium"
      >
        {loading ? "Submitting..." : "Sign Petition"}
      </button>
    </form>
  );
}
