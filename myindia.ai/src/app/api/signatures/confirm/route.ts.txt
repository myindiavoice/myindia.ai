import { NextRequest, NextResponse } from "next/server";
import { getSupabaseServer } from "@/lib/supabase/server";
import { verifyToken } from "@/lib/tokens";

export async function GET(req: NextRequest) {
  try {
    const token = req.nextUrl.searchParams.get("token");

    if (!token) {
      return NextResponse.json(
        { error: "Missing confirmation token" },
        { status: 400 }
      );
    }

    // Verify token
    const payload = verifyToken(token);
    if (!payload) {
      return NextResponse.json(
        { error: "Invalid or expired confirmation link" },
        { status: 400 }
      );
    }

    const supabase = getSupabaseServer();

    // Get petition slug for redirect
    const { data: petition } = await supabase
      .from("petitions")
      .select("slug")
      .eq("id", payload.petitionId)
      .single();

    if (!petition) {
      return NextResponse.json(
        { error: "Petition not found" },
        { status: 404 }
      );
    }

    // Call confirm_signature RPC function (atomic confirmation + increment)
    const { error } = await supabase.rpc("confirm_signature", {
      p_signature_id: payload.signatureId,
    });

    if (error) {
      console.error("Failed to confirm signature:", error);
      
      // Check if already confirmed
      if (error.message?.includes("already confirmed")) {
        // Redirect to petition page anyway
        const redirectUrl = new URL(
          `/petitions/${petition.slug}?confirmed=1`,
          req.url
        );
        return NextResponse.redirect(redirectUrl);
      }

      return NextResponse.json(
        { error: "Failed to confirm signature" },
        { status: 500 }
      );
    }

    // Redirect to petition page with success message
    const redirectUrl = new URL(
      `/petitions/${petition.slug}?confirmed=1`,
      req.url
    );
    return NextResponse.redirect(redirectUrl);
  } catch (error: any) {
    console.error("Error confirming signature:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
